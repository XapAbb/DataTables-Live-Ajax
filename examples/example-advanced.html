DatTables Live Ajax Plugin - Advanced Example

Description: Shows full compatibility with all ajax options, changes update interval and adds a function to pause ajax requests randomly

Other Examples
Source Used 
    $(document).ready(function() {
        $('#example').DataTable({
            ajax: {
                url: 'dataSrc.php',
                type: 'POST',
                data: { dataSrc: 'something'},
                dataSrc: 'something'
            },
            columns: [
                {
                    title: 'Name',
                    name:  'name',
                    data:  'name'
                },
                {
                    data:  'extn',
                    name:  'extn',
                    title: 'Extension'
                }
            ],
            rowId: 'emp_id',
            liveAjax: {
                // Update every 4.5 seconds
                interval: 4500,
                pause: function(){
                    // Randomly return TRUE to pause
                    return Math.random()<.5;
                }
            }
        });
    } );
